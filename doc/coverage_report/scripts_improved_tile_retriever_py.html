156def check_request(res, year, x_coord, y_coord): 

157 """try to fetch the tile's image from the tile server""" 

158 # this might either result in a HTTP 404 if the url is invalid 

159 # or in a MySQL error, if the connection to the database is lost 

160 try: 

161 file = urllib.request.urlopen(res) 

162 im = Image.open(file, 'r') 

163 finish_request(im, year, x_coord, y_coord) 

164 except Exception as e: 

165 print(e)

168def finish_request(im, year, x_coord, y_coord): 

169 """finish the request, by checking if the image is white or transparent""" 

170 pix_val = list(im.getdata()) 

171 if not check_transparent(pix_val): 

172 if not check_full_white(pix_val): 

173 print("image was good. saving...") 

174 save_tile(year, x_coord, y_coord) 

175 print("saved")

178def check_transparent(pix_val): 

179 """check if the image is transparent""" 

180 # check all 4 corners 

181 # if one of them has an alpha value, it is transparent 

182 # discard transparent images 

183 if len(pix_val[0]) < 4 and \ 

184 len(pix_val[255]) < 4 and \ 

185 len(pix_val[65280]) < 4 and \ 

186 len(pix_val[65535]) < 4: 

187 return False 

188 else: 

189 return True

192def check_full_white(pix_val): 

193 """check if the image if fully white""" 

194 # count how many of the check pixels are white 

195 count = 0 

196 for pix in pix_to_check: 

197 if pix_val[pix] == (255, 255, 255): 

198 count += 1 

199 # if at least 75% of counted pixels are white, consider the entire image to be white 

200 if count >= 575: 

201 return True 

202 else: 

203 return False
